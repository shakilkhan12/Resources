Form Helpers

Before using the form helper first you need to load the form helper in the controller and you can also load from the autoload file simple goto config folder and then click on autoload.php file. 

<?php 
/*
    * autoload helpers
*/ 
$autoload = [
   'helpers' => ['form']
];
 ?>

form_open

form open helper just produce the html form opening tag

<?php echo form_open(action, method, ["class" = "" "id" = ""]);
             
            //produces 
            <form action="" method="" class="" id="">
?>

form_close

form_close helper produce the html form closing tag

<?php echo form_close();
//produces
</form>
 ?>

form_input

form_input helper produce the html input field like text input field, email input field, password input field and file input field

<?php echo form_input(['type' => 'text','name' => 'full_name', 'id' => 'input-id', 'class' => 'form-control', 'placeholder' => 'Enter Full Name', 'value' =>'']);

//produces
<input type="text" name="full_name" id="input-id" class="form-control" placeholder="Enter Full Name" value="">

<?php echo form_input(['type' => 'email','name' => 'email', 'id' => 'input-id', 'class' => 'form-control', 'placeholder' => 'Enter Email', 'value' =>'']);

//produces
<input type="email" name="email" id="input-id" class="form-control" placeholder="Enter Email" value="">

<?php echo form_input(['type' => 'password','name' => 'password', 'id' => 'input-id', 'class' => 'form-control', 'placeholder' => 'Enter Password', 'value' =>'']);

//produces
<input type="password" name="password" id="input-id" class="form-control" placeholder="Enter Password" value="">

<?php echo form_input(['type' => 'file','name' => 'file', 'id' => 'input-id', 'class' => 'form-control']);

//produces
<input type="file" name="file" id="input-id" class="form-control">

 ?>

form_multipart

form_multipart helper produce the html form opening tag along with enctype attribute

<?php echo form_multipart(action, method, ["class" = "" "id" = ""]);

//produce
<form action="" method="" class="" id="" enctype="multipart/form-data"> 

 ?>

form_button

form_button helper proudce the html normal button bellow is the code

<?php echo form_button(["name" => "my-btn", "class" => "btn-default", "id" => "btn-id", "value" => "Save"]);

//produce 
<button type="button" name="my-btn" class="btn-default" id="btn-id">Save</button>


 ?>

form_submit

form_submit helper produce the html submit button

<?php echo form_submit(["name" => "my-btn", "class" => "btn-default", "id" => "btn-id", "value" => "Save"]);

//produce 
<input type="submit" name="my-btn" class="btn-default" id="btn-id" value="Save">


 ?>
